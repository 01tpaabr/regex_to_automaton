auxMedia([X], X, 1) :- !.
auxMedia([H | T], M, N) :-
    auxMedia(T, M2, N2),
    M is (M2 + H),
    N is (N2 + 1).

media([H | T], M, N) :-
    auxMedia([H | T], M1, N1),
    M is (M1 / N1),
    N is (N1 - 1).


desvioPadraoAux([], 0, _N, _L).

desvioPadraoAux([H | T], D, N, L) :-
    media(L, M1, N),
    desvioPadraoAux(T, D1, _N, L),
    D is ((D1 + ((H - M1) ** 2))).

desvioPadrao([H | T], D) :-
    desvioPadraoAux([H | T], D1, N, [H | T]),
    D is sqrt(D1/N).
